\begin{align*}
 L(aa^* \cup ba^*b)
  &= L(aa^*) \cup L(ba^*b) \cr
  &= L(a)L(a^*) \cup L(b)L(a^*)L(b) \cr
  &= \{a\}L(a)^* \cup \{b\}L(a)^*\{b\} \cr
  &= \{a\}\{a\}^* \cup \{b\}\{a\}^*\{b\} \cr
  &= \{ a^n \,|\, n \geq 1 \} \cup \{ ba^nb \,|\, n \geq 0 \}
 \end{align*}
