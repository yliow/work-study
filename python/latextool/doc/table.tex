             ('', '', '',)
            ],
            row_headings = ['0', '2', '5', '8'],
            col_headings = ['a', 'b', 'c'],
            topleft_heading = r'$\Delta X$',
            style='|c|l|r|c|',
           ))
'''.strip()

\end{python}
