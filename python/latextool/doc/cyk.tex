\subsubsection{Backgound}
You can set the background using the (row, col)
index values.
If it's set, the cell is also drawn with a larger
linewidth.
\begin{python}
s = r'''
from latextool_basic import *
m = [['?','$A$','$B$','','',''],
     ['','','','','',''],
     ['','','','','',''],
     ['','','','','',''],
     ['','','','','',''],
     ['','','','','',''],
     ]
p = Plot()
cyk(p, m, background={(0,0):'blue!10',
                      (0,1):'red!10',})
print(p)
'''
from latextool_basic import *
execute(s, print_source=True)
\end{python}
