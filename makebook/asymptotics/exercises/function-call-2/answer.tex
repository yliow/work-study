\begin{align*}
  T(n) &= O(n) \\
  \mathsc{Space}(n) &= O(n)
\end{align*}

The improved code is
\begin{Verbatim}[frame=single,fontsize=\small]
int h(const std::vector< int > & x)
{
    return x[0] + 1;
}

int g(const std::vector< int > & x)
{
    return x[0] - 1
}

void f(const std::vector< int > & x)
{
    if (x[0] < 0)
    {
        std::cout << g(x) << '\n';
    }
    else
    {
        std::cout << h(x) << '\n';
    }
}
\end{Verbatim}
with
\begin{align*}
  T(n) &= O(1) \\
  \mathsc{Space}(n) &= O(1)
\end{align*}
\qed
