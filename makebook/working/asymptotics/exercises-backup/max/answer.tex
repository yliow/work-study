(a)
\begin{Verbatim}[frame=single,fontsize=\small]
                             time    number of times
         M = x[0]            t1      1 
         i = 1               t2      1
LOOP:    if i >= n:          t3      n
            goto ENDLOOP     t4      1
         if x[i] <= M:       t5      n - 1
            goto ELSE        t6      n - 1
         M = x[i]            t7      0
ELSE:    i = i + 1           t8      n - 1
         goto LOOP           t9      n - 1
ENDLOOP:
\end{Verbatim}
\begin{align*}
T_b(n)
&= (t_3 + t_5 + t_6 + t_8 + t_9)n + (t_1 + t_2 + t_4 - t_5 - t_6 - t_8 - t_9)
&= An + B
\end{align*}
where $A, B$ are constants.
Therefore
\[
T_b(n) = O(n)
\]

(b)
\begin{Verbatim}[frame=single,fontsize=\small]
                             time    number of times
         M = x[0]            t1      1 
         i = 1               t2      1
LOOP:    if i >= n:          t3      n
            goto ENDLOOP     t4      1
         if x[i] <= M:       t5      n - 1
            goto ELSE        t6      0
         M = x[i]            t7      n - 1
ELSE:    i = i + 1           t8      n - 1
         goto LOOP           t9      n - 1
ENDLOOP:
\end{Verbatim}
\begin{align*}
T_w(n)
&= (t_3 + t_5 + t_7 + t_8 + t_9)n + (t_1 + t_2 + t_4 - t_5 - t_7 - t_8 - t_9)
&= Cn + D
\end{align*}
where $C, D$ are constants.
Therefore
\[
T_w(n) = O(n)
\]
