Rewrite the following using iterators:
\begin{console}[fontsize=\footnotesize]
template< typename T >
std::ostream & operator<<(std::ostream & cout,
                          const std::vector< T > & v)
{
    std::string delim = "";                            
    cout << '{';
    for (size_t i = 0; i < v.size(); ++i)
    {
        cout << delim << v[i];
        delim = ", ";
    }
    cout << '}';
    return cout
}  
\end{console}
