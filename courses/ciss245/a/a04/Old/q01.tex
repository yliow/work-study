 [Pointer parameters and pass-by-reference]
You already know that \cpp\ performs character,
integer and double inputs like this:
\begin{console}
char c = ' ';
int i = 0;
double d = 0;
std::cin >> c;
std::cin >> i;
std::cin >> d;
// or:
// std::cin >> c >> i >> d;
\end{console}

The father of \cpp, i.e. the C programming language,
performs character, integer, and double input like this:
\begin{console}
#include <cstdio>

int main()
{
    char c = ' ';
    int i = 0;
    double d = 0;
    
    scanf("%c", &c);  // char input for c
    scanf("%d", &i);  // int input for n
    scanf("%lf", &d); // double input for d
    std::cout << c << ' ' << i << ' ' << d << '\n';
    
    return 0;
}
\end{console}
You can compile that using a \cpp\ compiler.
But if you use a C compiler you have to change \verb!cstdio! to
\verb!stdio.h!.
C also prints things in a different way:
\begin{console}
#include <cstdio>

int main()
{
    char c = '#';
    int i = 0;
    double d = 0;
    char s[1024] = "hello world";
    
    printf("%c\n", c);
    printf("%d\n", i);
    printf("%lf\n", d);
    printf("%s\n", s);
    
    return 0;
}
\end{console}
