Now test it with a C++ hello world:
\begin{python}
from latextool_basic import *
s = r'''// file: tmp.cpp
#include <iostream>

int main()
{
    std::cout << "hello world" << std::endl;
    return 0;
}
'''.strip()
print console(s)
file('clang/tmp.cpp', 'w').write(s)
print "Compile and execute:"
print shell('clang++ tmp.cpp; ./a.out', dir='clang')
\end{python}
