Which method should be static?

\begin{consolethree}[escapeinside=||]
class C
{
public:
    C(char name[]) { strcpy(name_, name); }
    void print0() const
    {
        std::cout << "hello 0\n";
    }
    void print1() const
    {
        std::cout << "hello " << name_ << std::endl;
    }
private:
    char name_[100];
};

int main()
{
    C c("john");
    c.print0();
    c.print1();
    return 0;
}
\end{consolethree}
