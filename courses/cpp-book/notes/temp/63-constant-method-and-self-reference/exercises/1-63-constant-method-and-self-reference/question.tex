\begin{ex}
Which parameter should be made \texttt{const}?

\begin{consolethree}[escapeinside=||]
class Int
{
public:
    Int(int x0)
        : x_(x0)
    {}
    void incrementBy(Int & a)
    {
        x_ += a.x_;
    }
    void increment(Int & a)
    {
        a.x_ += x_;
    }
    int get()
    {
        return x_;
    }
private:
    int x_;
};

int main()
{
    Int a(1), b(5);
    a.incrementBy(b);
    a.increment(b);
    std::cout << a.get() << ' ' << b.get() << '\n';
    return 0;
}
\end{consolethree}
\end{ex}
