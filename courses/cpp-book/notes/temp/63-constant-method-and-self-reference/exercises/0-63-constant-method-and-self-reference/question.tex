\begin{ex}
Which parameters in \texttt{j,k} can be made constant?

\begin{consolethree}[escapeinside=||]
class C
{
public:
    void f(int a)
    {
        x_ = a;
    }
    int g(int a)
    {
        return y_ + a;
    }
    void h(C & c)
    {
        f(c.x_);
    }
    void i(C & c)
    {
        c.x_ = g(5);
    }
private:
    int x_, y_;
};

void j(C & c, C & d)
{
    c.h(d);
}

void k(C & c, C & d)
{
    c.i(d);
}

int main()
{
    C c, d;
    j(c, d);
    k(c, d);
    return 0;
}
\end{consolethree}
\end{ex}
