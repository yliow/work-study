\begin{ex}
Which methods should be constant? Which parameters
should be constant?

\begin{consolethree}[escapeinside=||]
class C
{
public:
    void m(C & a, C & b)
    {
        x_ = a.x_;
        y_ = b.y_;
    }
    void m(C & a, C & b)
    {
        x_ = a.x_;
        b.y_ = y_;
    }
    void m(C & a, C & b)
    {
        a.x_ = x_;
        y_ = b.y_;
    }
    void m(C & a, C & b)
    {
        a.x_ = x_;
        b.y_ = y_;
    }
private:
    int x_, y_;
};
\end{consolethree}
\end{ex}
