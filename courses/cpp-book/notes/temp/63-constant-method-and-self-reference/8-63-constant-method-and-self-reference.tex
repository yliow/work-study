\sectionthree{Yet more examples on returning *this as a reference (DIY)}

\begin{consolethree}[escapeinside=||]
class Point2d
{
public:
    Point2d & set_x(int x) { x_ = x; return *this; }
    Point2d & set_y(int y) { y_ = y; return *this; }
    void print()
    {
        std::cout << x_ << ',' << y_ << '\n';
    }
private:
    int x_, y_;
};

int main()
{
    Point2d p;
    Point2d & q = p.set_x(1);
    q.set_y(2);
    p.print();
    return 0;
}
\end{consolethree}
