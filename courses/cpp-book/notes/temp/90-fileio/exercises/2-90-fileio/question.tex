\begin{consolethree}[escapeinside=||]
#include <iostream>
#include <fstream>

int main()
{   
    std::ifstream f("xyz.txt", std::ios::in);
    if (|\textbf{f.is\_open()}|)
    {    
         std::cout << "error opening file xyz.txt"
                   << std::endl;
    }
    else
    {   
         // do what you want to do ...
         f.close();
    }
    
    return 0;
}
