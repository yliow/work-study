\begin{consolethree}[escapeinside=||]

#include <iostream>

int sum(int x[], int size)

{

    int s = 0;

    for (int i = 0; i < size; i++)

    {

        s += x[i];

    }

    return s;

}

double avg(int x[], size)

{

    return double(sum(x, size)) / size;

}

int main()

{

    int a[] = {1, 3, 5, 7, 2, 4, 6, 8};

    std::cout << avg(a + 4, 3) << '\n';

    return 0;

}

\end{consolethree}
