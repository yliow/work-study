\tinysidebar{\debug{exercises/{49-27-pointers-and-arrays/question.tex}}}Write a program that implements a stack of integers using dynamic arrays. Test it thoroughly! You can test your stack implementation with the following:



\begin{consolethree}[escapeinside=||]

int main()

{

    int * stack = NULL;

    int stack_size = 0;

    println(stack, stack_size); // Get {}

    push(&stack, &stack_size, 5);

    println(stack, stack_size); // Get {5}

    push(&stack, &stack_size, 3);

    println(stack, stack_size); // Get {5, 3}

    push(&stack, &stack_size, 9);

    println(stack, stack_size); // Get {5, 3, 9}

    int x;

    x = peek(stack, stack_size);

    std::cout << x << std::endl; // Get 9

    println(stack, stack_size); // Get {5, 3, 9}

    pop(&stack, &stack_size);

    println(stack, stack_size); // Get {5, 3}

    return 0;

}

\end{consolethree}
