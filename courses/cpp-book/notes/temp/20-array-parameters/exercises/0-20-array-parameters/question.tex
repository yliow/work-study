\begin{ex}
Write a function that returns the product of all
doubles in the array that is passed in to the function.

\begin{consolethree}[escapeinside=||]
_____________ product(double x[], int x_size)
{
}

int main()
{
    double a[] = {1, 2, 3};
    std::cout << product(a, 1) << '\n'; // 1
    std::cout << product(a, 2) << '\n'; // 2
    std::cout << product(a, 3) << '\n'; // 6
    return 0;
}
