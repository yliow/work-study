\begin{ex}
Which function is called (or is there an error)?

\begin{consolethree}[escapeinside=||]
void f(int x) {}
void f(double x) {}

int main()
{
    f(1);
}
\end{consolethree}

Check by running this version:

\begin{consolethree}[escapeinside=||]
#include <iostream>

void f(int x) { std::cout << "f(int)\n"; }
void f(double x) { std::cout << "f(double)\n"; }

int main()
{
    f(1);
}
\end{consolethree}
\end{ex}
