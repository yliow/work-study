\sectionthree{Pre and post operators}

\begin{consolethree}[escapeinside=||]
class Int
{
public:
    Int(int x=0)
        : x_(x)
    {}
    void operator++()
    {
        std::cout << "pre\n"; ++x_;
    }
    void operator++(int a)
    {
        std::cout << "post " << a << "\n"; x_++;
    }
    int get() { return x_; }
private:
    int x_;
};
\end{consolethree}

\begin{consolethree}[escapeinside=||]
int main()
{
    C x(4);
    ++x;
    std::cout << x.get() << "\n";
    x++;
    std::cout << x.get() << "\n";
    return 0;
}
\end{consolethree}

C++ translates \textit{++x} and \textit{x++} as:

\textit{++x} is the same as \textit{x.operator++()}

\textit{x++} is the same as \textit{x.operator++(0)}

i.e., for post operators, C++ calls operator with a value of 0.
