\begin{consolethree}[escapeinside=||]
#include <iostream>

class Int
{
public:
    Int(int x)
        : x_(x)
    {
        std::cout << "Int\n";
    }
    Int operator+(const Int & i)
    {
        return Int(x_ + i.x_);
    }
private:
    int x_;
};
\end{consolethree}
