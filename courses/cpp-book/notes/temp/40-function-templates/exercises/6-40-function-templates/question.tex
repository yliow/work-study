\begin{consolethree}[escapeinside=||]

#include <iostream>



template < typename T >

T max(T x[], int size)

{

    T m = x[0];

    for (int i = 1; i < size; i++)

    {

        if (m < x[i]) m = x[i];

    }

    return m;

}



int main()

{

    int x[10] = {3, 1, 5, 6, 2, 8, 9, 3, -2, 3};

    std::cout << max< int >(x, 10) << std::endl;



    double y[8] = {3.1, 1.8, 5.2, 6.7,

                   8.4, 9.1, 3.6, -2.7};

    std::cout << max< double >(y, 10) << std::endl;

    return 0;

}

\end{consolethree}
