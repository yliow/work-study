\tinysidebar{\debug{exercises/{19-40-function-templates/question.tex}}}Write a template function with specialization(s) to

print an array.



\begin{consolethree}[escapeinside=||]

int x[] = {2, 3, 5};

array_println< int >(&x[0], &x[3]);

// prints {2, 3, 5} and '\n'



double y[] = {2.1, 3.1, 5.1};

array_println< double >(&y[0], &y[3]);

// prints {2.1, 3.1, 5.1} and '\n'



char z[] = "abcde";

array_println< char >(&z[0], &z[5]);

// prints abcde and '\n'

\end{consolethree}
