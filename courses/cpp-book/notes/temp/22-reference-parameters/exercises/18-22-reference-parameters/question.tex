\begin{ex}
Given this code

\begin{consolethree}[escapeinside=||]
#include <iostream>

int f(int & a, double & b)
{
    a++;
    b += a;
    return a * b
}

int main()
{
    int x = 42;
    double y = 1;
    int z = f(x, y);
    std::cout << x << ' ' << y << ' ' << z
              << std::endl;
    return 0;
}
\end{consolethree}

Rewrite it into this form:

\begin{consolethree}[escapeinside=||]
#include <iostream>

// prototype of f() here

int main()
{
    int x = 42;
    double y = 1;
    int z = f(x, y);
    std::cout << x << ' ' << y << ' ' << z
              << std::endl;
    return 0;
}

int f(int & a, double & b)
{
    a++;
    b += a;
    return a * b
}
\end{consolethree}

Run your program to make sure that you did it correctly.
\end{ex}
