\tinysidebar{\debug{exercises/{17-22-reference-parameters/question.tex}}}After declaring a reference can the reference change the variable it's referring to? Does = work?



\begin{consolethree}[escapeinside=||]

int x = 1;

int y = 2;

int & z = x; // make z refer to x

z = 0;

std::cout << x << ' ' << y << ' ' << z << std::endl;



z = y; // make z refer to y ... can we?

z = -1;

std::cout << x << ' ' << y << ' ' << z << std::endl;

\end{consolethree}
