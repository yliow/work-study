#include <iostream>

void println(int x[], int len)
{
    for (int i = 0; i < len; i++)
    {
        std::cout << x[i] << ' ';
    }
    std::cout << std::endl;
}

void arrayAssign(int y[], |\EMPHASIZE{int \& ylen}|,
                 int x[], int xlen)
{
    ylen = xlen;
    for (int i = 0; i < xlen; i++)
    {
        y[i] = x[i];
    }
}

int main()
{
    int x[] = {1, 2, 3};
    int y[100];
    int ylen = 0;
    arrayAssign(y, ylen, x, 3);
    println(y, ylen);
    return 0;
}
