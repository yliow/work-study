\sectionthree{Coding style for nested \texttt{if}-\texttt{else}}

NOT SO GOOD:\\
\begin{console}
if (x < 5)
{
    foo = 42;
}
else
{
     if (x < 8)
     {
         foo = 43;
     }
     else
     {
         if (x == 9)
         {
             foo = 44;
         }
         else
         {
             foo = 45;
         }
     }
}
\end{console}
GOOD:\\
\begin{console}
if (x < 5)
{
    foo = 42;
}
else if (x < 8)
{
    foo = 43;
}
else if (x == 9)
{
    foo = 44;
}
else
{
    foo = 45;
}
\end{console}
