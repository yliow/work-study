Fixit time \ldots (you are only allowed to delete one statement.)

\begin{consolethree}[escapeinside=||]
#include <iostream>

void average(double x[])
{
    double x = 0;
    double * px = &x;
    for (int i = 0; i < 10; ++i)
    {
        *px += x[i];
    }

    x = *px / 10;
    delete px;
    return x;
}

int main()
{
    double x[5] = {1.1, 2.1, 2.3, 3.2, 3.4};
    std::cout << average(x) << std::endl;
    return 0;
}
\end{consolethree}
\end{ex}
