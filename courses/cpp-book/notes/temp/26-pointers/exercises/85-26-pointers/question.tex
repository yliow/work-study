#include <iostream>
#include <cstddef>

void swap(int * p, int * q)
{
    int * t = p;
    p = q;
    q = t;
}

int main()
{
    int x = 0, y = 42;
    int * p = &x, * q = &y;

    std::cout << *p << ' ' << *q << '\n'; // 0 42

    swap(p, q); // p should point to x

                // q should point to y

                // Do you get 42 0?

    std::cout << *p << ' ' << *q << '\n';
    return 0;
}
