\begin{consolethree}[escapeinside=||]

int x = 0, y = 1;

int * p = &x;

std::cout << *p << '\n';

p = &y;

std::cout << *p << '\n'; 

\end{consolethree}

Everything works just fine. Now make \texttt{p} a constant pointer like this:



\begin{consolethree}[escapeinside=||]

int x = 0, y = 1;

int * |\EMPHASIZE{const}| p = &x;

std::cout << *p << '\n';

p = &y; // compiler complaints!!!

std::cout << *p << '\n'; 

\end{consolethree}

Get it?
