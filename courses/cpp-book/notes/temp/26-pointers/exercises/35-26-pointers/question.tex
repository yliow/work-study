\begin{consolethree}[escapeinside=||]
#include <iostream>

int h(int * z)
{   
    return *z;
}

int * g(int * y)
{   
    return y;
}

int * f(int * x)
{   
    return (&x == NULL ? NULL : g(x));
}

int main()
{   
    int a = 21;
    std::cout << *f(&a) + h(&a) << std::endl;
    return 0;
}
\end{consolethree}
