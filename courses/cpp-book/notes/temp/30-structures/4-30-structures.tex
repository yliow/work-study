\begin{consolethree}[escapeinside=||]
#include <iostream>

struct Student
{
    char lastname[50];
    char firstname[50];
    double gpa;
};

int main()
{
    Student s[3] = {{"Doe", "John", 3.55},
                    {"Smith", "Jane", 3.75},
                    {"Woo", "Bob", 3.25}};

    for (int i = 0; i < 3; ++i)
    {
        std::cout << s[i].firstname << ' '
                  << s[i].lastname << ": "
                  << s[i].gpa << '\n';
    }

    return 0;
}
