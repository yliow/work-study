#include <iostream>



int main()

{   

     int answer, i, first, last, sum, term;

     std::cout << "Find sum? (1 -- yes or 0 - no) ";

     std::cin >> answer;

     

     if (answer == 1)

     {    

          std::cout << "First int:";

          std::cin >> first;

          std::cout << "Last int:"

          std::cin >> last;



          sum = 0;

          for (i = first; i <= last; ++i)

          {    

               term = i * i * i * i;

               sum += term;

          }

          std::cout << "Sum:" << sum << std::endl;

     }



     return 0;

}
